 15class Source(Container):
 16    """Manages an astronomical source/region properties and data.
 17
 18    Attributes:
 19      name: name of the source.
 20      config_file: configuration file name.
 21      config: configuration file of the source.
 22      subsources: region subsources (if any).
 23      log: logging manager.
 24      _data: the data belonging to the source.
 25    """
 26    log = get_logger(__name__, filename=__package__+'.log')
 27
 28    def __init__(self,
 29                 name: Optional[str] = None,
 30                 config_file: Optional[Union[Path, str]] = None,
 31                 config: Optional[ConfigParserAdv] = None):
 32        """Creates a new Source.
 33
 34        Args:
 35          name: optional; the name of the source.
 36          config_file: optional; configuration file name.
 37          config: optional; `ConfigParserAdv` object.
 38        """
 39        # Initialize
 40        if name is not None:
 41            self.log.info('Initializing source: %s', name)
 42        else:
 43            self.log.info('Initializing source from configuration')
 44        super().__init__(name, config_file=config_file, config=config,
 45                         default_section='INFO')
 46
 47        # Initiate subsources
 48        self.subsources = {}
 49        self.load_subsources()
 50
 51    def __str__(self):
 52        """String representation."""
 53        lines = [f'{self.name}', '-'*len(self.name)]
 54        for opt, val in self.config.items('INFO'):
 55            lines.append(f'{opt} = {val}')
 56        if self.subsources:
 57            lines.append('Subsources:')
 58            lines.append(','.join(f' {val.name} (key)'
 59                                  for key, val in self.subsources.items()))
 60        if self._data:
 61            lines.append('Loaded data:')
 62            for key in self._data:
 63                lines.append(f'\t{key}')
 64        lines.append('-'*len(self.name))
 65        return '\n'.join(lines)

 67    @classmethod
 68    def from_values(cls, name: str, **kwargs):
 69        """Generates a new instance storing the input arguments."""
 70        # Store values
 71        config = ConfigParserAdv()
 72        config.read_dict({'INFO': {'name': name}})
 73        for key, val in kwargs.items():
 74            if hasattr(val, 'unit'):
 75                unit = f'{val.unit}'.replace(' ', '')
 76                config['INFO'][key] = f'{val.value} {unit}'
 77            else:
 78                config['INFO'][key] = f'{val}'
 79
 80        return cls(config=config)

121    def load_subsources(self) -> None:
122        """Load the subsources in `Source` configuration."""
123        for section, conf in self.config.items():
124            if not conf.get('type', fallback='') == 'subsource':
125                continue
126
127            self.subsources[section] = SubSource.from_config_proxy(conf,
128                                                                   name=section)

130    def load_data(self, section: str,
131                  file_name: Optional[Union[Path, str]] = None,
132                  **kwargs) -> None:
133        """Load the data.
134
135        It uses the classes registered in `REGISTERED_CLASSES` to identify
136        which type of data it has to load. The data information (e.g. file
137        name) has to be in the source configuration file.
138
139        Args:
140          section: the data section to be loaded.
141          file_name: optional; overwrites the config file name.
142          kwargs: optional; additional arguments for loader function.
143        """
144        # File name
145        if file_name is not None:
146            data_file = Path(file_name).expanduser()
147        else:
148            data_file = self.config.getpath(section, 'file', fallback=None)
149        if data_file is None:
150            raise ValueError(f'Could not load data for {section}')
151
152        # Load data
153        dtype = self.config[section]['type'].lower()
154        kwargs_load = kwargs_from_config(self.config[section])
155        kwargs_load.update(kwargs)
156        self._data[section] = load_data_by_type(data_file,
157                                                dtype,
158                                                REGISTERED_CLASSES,
159                                                **kwargs_load)
160
161        # Update config
162        super().load_data(section, file_name=file_name)
163
164    def load_all_data(self):
165        """Load all the data in the configuration file."""
166        for section in self.config.sections():
167            if section == 'INFO' or 'type' not in self.config[section]:
168                continue
169            elif self.config[section]['type'] == 'subsource':
170                continue
171            else:
172                self.logger.info(f'Loading: {section}')
173                self.load_data(section)
174
175    def load_config(self,
176                    config_file: Path,
177                    update_from: Optional[ConfigParserAdv] = None,
178                    default_section: str = 'DEFAULT') -> None:
179        """Load a configuration file.
180
181        Args:
182            config_file (str): name of the configuration file
183        """
184        super().load_config(config_file, update_from=update_from,
185                            default_section=default_section)
186
187        # Check INFO section
188        if 'INFO' not in self.config:
189            self.log.warning(('Source does not have INFO section: '
190                              'some functions will not work'))
191
192    def write(self, filename: Optional[Path] = None) -> None:
193        """Write configuration file to disk."""
194        # Store subsources in config
195        if self.subsources:
196            for key, val in self.subsources.items():
197                if key not in self.config:
198                    self.config[key] = val.to_dict()
199
200        # Write
201        super().write(filename=filename)
202
203    def get_data_sections(self) -> Sequence:
204        return super().get_data_sections(tuple(REGISTERED_CLASSES.keys()))

206class SubSource(object):
207    """Class for storing individual source information.
208
209    Some types of information have default types:
210
211    - `ra`, `dec` and `frame` are stored as `SkyCoord`.
212    - `radius` is stored as `astropy.units.Quantity`.
213
214    If only `ra` and `dec` are given, then ICRS is assumed as `frame`.
215
216    Attributes:
217      name: subsource name.
218      info: source information.
219    """
220
221    def __init__(self, name: str, **info):
222        self.name = name
223        self.info = info
224
225    @classmethod
226    def from_config_proxy(cls,
227                          parser: ConfigParserAdv,
228                          name: Optional[str] = None):
229        """Create a new subsource from a `ConfigParser` proxy.
230
231        Args:
232          parser: configuration parser proxy.
233          name: optional; subsource name.
234        """
235        name = parser.get('name', fallback=name)
236
237        return cls.from_dict(parser, name=name)
238
239    @classmethod
240    def from_dict(cls,
241                  data: dict,
242                  name: Optional[str] = None):
243        """Create a new subsource from a dictionary.
244
245        Args:
246          data: configuration parser proxy.
247          name: optional; subsource name.
248        """
249        if name is None:
250            name = data.pop('name', name)
251        ignore_keys = ['type', 'name']
252        info = {}
253        position = {'frame': 'icrs'}
254        for key, val in data.items():
255            if key in ignore_keys:
256                continue
257            elif key in ['ra', 'dec', 'frame']:
258                position[key] = val
259            elif key in ['radius']:
260                info[key] = u.Quantity(val)
261            else:
262                info[key] = val
263
264        if 'ra' in position and 'dec' in position:
265            info['position'] = SkyCoord(**position)
266
267        return cls(name, **info)
268
269    @property
270    def position(self):
271        """Position of the subsource."""
272        return self.info.get('position')
273
274    def to_dict(self) -> dict:
275        """Convert the values back to a dictionary with string values."""
276        props = {'name': self.name, 'type': 'subsource'}
277        for key, val in self.info.items():
278            if key == 'position':
279                props['ra'], props['dec'] = val.to_string('hmsdms').split()
280                props['frame'] = val.frame.name
281            elif hasattr(val, 'unit'):
282                props[key] = f'{val.value} {val.unit}'
283            else:
284                props[key] = val
285
286        return props

Some types of information have default types:

  • ra, dec and frame are stored as SkyCoord.
  • radius is stored as astropy.units.Quantity.

If only ra and dec are given, then ICRS is assumed as frame.
